# Add structured per-round trace output

def run_simulation(mechanism, behaviour, T, seed=42):
    rng = np.random.default_rng(seed)
    traces = []

    for t in range(T):
        reports   = behaviour.get_reports(t, rng)
        sigma     = mechanism.skill.update(t)
        deposits  = mechanism.deposit.compute(sigma, wealth)
        wagers    = mechanism.influence.compute(deposits, sigma)
        weights   = wagers / wagers.sum()
        aggregate = mechanism.aggregation.pool(reports, weights)
        y_t       = dgp.realise(rng)
        scores    = mechanism.scoring.crps(reports, y_t)
        payouts   = mechanism.settlement.pay(deposits, wagers, scores)
        profits   = payouts - deposits
        wealth   += profits

        # ← NEW: structured trace per round
        traces.append({
            "t": t, "y_t": y_t,
            "sigma": sigma.tolist(),
            "deposits": deposits.tolist(),
            "wagers": wagers.tolist(),
            "weights": weights.tolist(),
            "aggregate": aggregate,
            "scores": scores.tolist(),
            "profits": profits.tolist(),
            "wealth": wealth.tolist(),
            "hhi": float((weights**2).sum()),
            "n_eff": float(1 / (weights**2).sum()),
        })

    return traces

const STEPS = [
  { key: 'reports',    sym: 'q_i',  label: 'Report'    },

  { key: 'weights',    sym: 'w̃_i',  label: 'Weight'    },

  { key: 'scores',     sym: 's_i',  label: 'Score'     },
  { key: 'profits',    sym: 'π_i',  label: 'Profit'    },
] as const

export function RoundInspector({ trace }: { trace: RoundTrace }) {
  const [step, setStep] = useState(0)
  const [pinned, setPinned] = useState<number | null>(null)
  const active = STEPS[step]

  return (
    <div>
      <Ribbon steps={STEPS} active={step} onSelect={setStep} />
      <ForecasterTable
        trace={trace}
        highlightKey={active.key}
        pinnedRow={pinned}
        onRowClick={setPinned}
      />
    </div>
  )
}

// Returns delta metrics between two mechanism configs

export function useCompare(baseRuns, altRuns) {
  return useMemo(() => {
    const last = (arr) => arr[arr.length - 1]

    return {
      dCRPS:      meanCRPS(altRuns) - meanCRPS(baseRuns),
      dNeff:      meanNeff(altRuns) - meanNeff(baseRuns),
      dHHI:       meanHHI(altRuns)  - meanHHI(baseRuns),
      dUtility:   meanUtility(altRuns) - meanUtility(baseRuns),
      dWealthGini:gini(last(altRuns).wealth) - gini(last(baseRuns).wealth),
    }
  }, [baseRuns, altRuns])
}

// Usage in DeltaBadge component:
function DeltaBadge({ label, delta, lowerIsBetter = false }) {
  const good = lowerIsBetter ? delta < 0 : delta > 0
  return (
    <span style={{ color: good ? '#085041' : '#993C1D' }}>
      {label}: {delta > 0 ? '+' : ''}{delta.toFixed(3)}
    </span>
  )
}
